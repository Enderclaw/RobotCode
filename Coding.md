void pre_auton()
{
  bStopTasksBetweenModes = true;
  
  SensorValue[QuadLeft] = 0;
  SensorValue[QuadRight] = 0;
}

task autonomous()
{
	motor[Claw] = 127; //open claw
	wait10Msec(250);

	motor[MainLift] = 127; // raise claw
  	wait10Msec(500);

  	motor[Claw] = -127; //close claw
  	wait10Msec(100);

	while(SensorValue[QuadRight] < 1800) //drive forward for 5 rotations
	{
		if(SensorValue[QuadLeft] < SensorValue[QuadRight])
		{
			motor[FDL] = 50;
			motor[FDL2] = 50;
			motor[FDR] = 63;
			motor[FDR2] = 63;
		}

		if(SensorValue[QuadLeft] > SensorValue[QuadRight])
		{
			motor[FDL] = 63;
			motor[FDL2] = 63;
			motor[FDR] = 50;
			motor[FDR2] = 50;
		}

		if(SensorValue[QuadLeft] == SensorValue[QuadRight])
		{
			motor[FDL] = 63;
			motor[FDL2] = 63;
			motor[FDR] = 63;
			motor[FDR2] = 63;
		}
	}

	motor[MainLift] = -127; //lower claw onto stationary high stack
  	wait10Msec(250);

  	motor[Claw] = 127; //open claw
  	wait10Msec(250);

  	while(SensorValue[QuadRight] > 720) //drive away from high stack
	{
		if(SensorValue[QuadLeft] < SensorValue[QuadRight])
		{
			motor[FDL] = -50;
			motor[FDL2] = -50;
			motor[FDR] = -63;
			motor[FDR2] = -63;
		}

		if(SensorValue[QuadLeft] > SensorValue[QuadRight])
		{
			motor[FDL] = -63;
			motor[FDL2] = -63;
			motor[FDR] = -50;
			motor[FDR2] = -50;
		}

		if(SensorValue[QuadLeft] == SensorValue[QuadRight])
		{
			motor[FDL] = -63;
			motor[FDL2] = -63;
			motor[FDR] = -63;
			motor[FDR2] = -63;
		}
	}
}

task usercontrol()
{
  while (true)
  {
  	motor[FDL] = vexRT[Ch3];
  	motor[FDL2] = vexRT[Ch3];

  	motor[FDR] = vexRT[Ch2];
  	motor[FDR2] = vexRT[Ch2];

  	if(vexRT[Btn5U] == 1)
  	{
  		motor[MainLift] = 127;
  	}

  	else if(vexRT[Btn5D] == 1)
  	{
  		motor[MainLift] = -127;
	}

  	else
	{
		motor[MainLift] = 0;
  	}

  	if(vexRT[Btn6U] == 1)
  	{
  		motor[Claw] = 127;
  	}

  	else if(vexRT[Btn6D] == 1)
  	{
  		motor[Claw] = -127;
  	}

  	else
  	{
  		motor[Claw] = 0;
  	}
  	
  	if(vexRT[Btn8U] == 1)
  	{
  		motor[MGLL] = 127;
  		motor[MGLR] = 127;
  	}
  	else if(vexRT[Btn8D] == 1)
  	{
  		motor[MGLL] = -127;
  		motor[MGLR] = -127;
  	}
  	else
  	{
  		motor[MGLL] = 0;
  		motor[MGLR] = 0;
  	}
    }
}
