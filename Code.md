task autonomous()
{
	while(SensorValue[QuadRight] < 270 || SensorValue[QuadLeft] > -270)
	{
		if(SensorValue[QuadRight] < 270)
		{
			motor[FDR] = 63;
		}
		else
		{
			motor[FDR] = 0;
		}
		if(SensorValue[QuadLeft] > -270)
		{
			motor[FDL] = -63;
		}
		else
		{
			motor[FDL] = 0;
		}
	}

	while(SensorValue[QuadRight] > -1800)
	{
		if(SensorValue[QuadLeft] < SensorValue[QuadRight])
		{
			motor[FDL] = -50;
			motor[FDR] = -63;
		}
		if(SensorValue[QuadLeft] > SensorValue[QuadRight])

			motor[FDL] = -63;

		}
